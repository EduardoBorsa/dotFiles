#To enable mousepad in i3
Create this file
/etc/X11/xorg.conf.d/30-touchpad.conf

With this content:

Section "InputClass"
        Identifier "10"
        MatchIsTouchpad "on"
        Driver "libinput"
    Option "Tapping" "on"
    Option "AccelSpeed" "0.8"

EndSection
